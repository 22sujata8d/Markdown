    def wish_greeting(name, lang):
        if lang == "Spanish":
          print("Hola, Mucho Gusto "+name)
        elif lang == "English":
          print("Hello, Nice to meet you "+name)
        else:
          print("Lang not found in code")

    For code snippet :      Give four space and start writing the code.
